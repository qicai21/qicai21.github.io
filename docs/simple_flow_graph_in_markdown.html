st=>start: start
click_start=>operation: 点击抢购
success=>end: 抢购成功
fail=>end: 抢购失败
login=>end: m.jd.com/login
retry=>condition: 再次抢购
divide=>operation: 分流服务器
captcha=>operation: 验证服务器: 登记的sn和时间,生成rid
seckill=>operation: 提交服务器
sn_fetch=>condition: 用thor计算sn
sn_status=>condition: sn已生效
success_check=>condition: 抢购成功:rid,data有效,有货

st->sn_fetch(yes)->click_start->divide->sn_status(yes)->captcha->seckill->success_check(yes)->success
st->sn_fetch(no)->fail
